% Experimental workflow figure (Figure 2) - standalone TikZ
% Include in main paper with \input{fig_workflow.tex}
\begin{figure}[t]
\centering
\resizebox{0.48\textwidth}{!}{%
\begin{tikzpicture}[
    node distance=0.5cm and 0.6cm,
    box/.style={rectangle, draw=#1, fill=#1!8, rounded corners=2pt,
                minimum width=2.4cm, minimum height=0.7cm,
                font=\small, align=center, line width=0.7pt},
    widebox/.style={rectangle, draw=#1, fill=#1!8, rounded corners=2pt,
                    minimum width=5.4cm, minimum height=0.7cm,
                    font=\small, align=center, line width=0.7pt},
    arrow/.style={-{Stealth[length=4pt]}, line width=0.7pt, #1},
    label/.style={font=\scriptsize\itshape, text=gray!60!black},
]

% Original dataset
\node[widebox=blue!50] (orig) {Original Dataset};

% Train-test split
\node[box=blue!50, below left=0.8cm and 0.6cm of orig] (train) {Training Set};
\node[box=blue!50, below right=0.8cm and 0.6cm of orig] (test) {Test Set};

\draw[arrow=blue!50] (orig.south) -- ++(0,-0.25) -| (train.north)
    node[pos=0.0, right, label] {Train-Test Split};
\draw[arrow=blue!50] (orig.south) -- ++(0,-0.25) -| (test.north);

% Feature split - training side
\node[box=blue!70, below left=0.8cm and 0.3cm of train] (basefeat) {Base\\Features};
\node[box=green!60!black, below right=0.8cm and 0.3cm of train] (extfeat) {Extended\\Features};

\draw[arrow=blue!50] (train.south) -- ++(0,-0.25) -| (basefeat.north)
    node[pos=0.0, right, label] {Feature Split};
\draw[arrow=blue!50] (train.south) -- ++(0,-0.25) -| (extfeat.north);

% Feature split - test side
\node[box=blue!70, below left=0.8cm and 0.3cm of test] (testbase) {Base Test\\Cases};
\node[box=green!60!black, below right=0.8cm and 0.3cm of test] (testext) {Extended\\Test Cases};

\draw[arrow=blue!50] (test.south) -- ++(0,-0.25) -| (testbase.north)
    node[pos=0.0, right, label] {Feature Split};
\draw[arrow=blue!50] (test.south) -- ++(0,-0.25) -| (testext.north);

% Phase labels
\node[left=0.1cm of basefeat, label, rotate=90, anchor=south] {Phase 1};
\node[right=0.1cm of extfeat, label, rotate=-90, anchor=south] {Phase 2};

% Models
\node[box=blue!70, below=1.0cm of basefeat] (basemodel) {Base Model};
\node[box=purple!60, below=1.0cm of extfeat, minimum width=2.8cm] (extmodel) {Extended Model\\{\footnotesize (w/ Adaptive Pruning)}};

\draw[arrow=blue!70] (basefeat) -- (basemodel);
\draw[arrow=green!60!black] (extfeat) -- (extmodel);

% Transfer arrow
\draw[-{Stealth[length=4pt]}, line width=0.9pt, dashed, purple!70]
    (basemodel.east) -- (extmodel.west)
    node[midway, above, font=\scriptsize\itshape, text=purple!70] {Transfer};

% Evaluation
\node[box=purple!60, below=1.5cm of test, minimum width=3.2cm] (eval)
    {Model Evaluation\\{\footnotesize (AUC Comparison)}};

\draw[arrow=blue!70] (testbase.south) |- ([yshift=0.3cm]eval.west) -- (eval.west);
\draw[arrow=green!60!black] (testext.south) |- ([yshift=-0.3cm]eval.east) -- (eval.east);

\end{tikzpicture}%
}
\caption{Experimental workflow. The original dataset is split into training and test sets, then features are partitioned into base and extended sets. Phase~1 trains a base model; Phase~2 transfers knowledge and trains with extended features. Both test populations are evaluated separately against the baseline for AUC comparison.}
\label{fig:experimental_workflow}
\end{figure}
